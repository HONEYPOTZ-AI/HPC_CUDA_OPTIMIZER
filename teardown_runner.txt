#!/usr/bin/env bash
# =============================================================================
# teardown_runner.sh  —  De-register and clean up the HPC GitHub Actions runner
#
# Usage:
#   bash teardown_runner.sh [--rundir <install_dir>] [--token <removal_token>]
#                           [--force] [--keep-workdir]
#
# Options:
#   --rundir <path>   Runner installation directory (default: /opt/gh-runner)
#   --token  <tok>    GitHub removal token (fetched via API if GH_PAT is set)
#   --force           Skip confirmation prompt
#   --keep-workdir    Do not delete the work directory
#
# =============================================================================
set -euo pipefail

RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[1;33m'
CYAN='\033[0;36m'; RESET='\033[0m'
info()    { echo -e "${CYAN}[teardown]${RESET} $*"; }
success() { echo -e "${GREEN}[teardown][OK]${RESET} $*"; }
warn()    { echo -e "${YELLOW}[teardown][WARN]${RESET} $*"; }
die()     { echo -e "${RED}[teardown][ERR]${RESET} $*" >&2; exit 1; }

# ── Defaults ──────────────────────────────────────────────────────────────────
RUNNER_INSTALL_DIR="${RUNNER_INSTALL_DIR:-/opt/gh-runner}"
REMOVAL_TOKEN="${REMOVAL_TOKEN:-}"
GH_PAT="${GH_PAT:-}"
FORCE=false
KEEP_WORKDIR=false
SERVICE_NAME="github-runner-hpc"

while [[ $# -gt 0 ]]; do
  case "$1" in
    --rundir)       RUNNER_INSTALL_DIR="$2"; shift 2 ;;
    --token)        REMOVAL_TOKEN="$2";      shift 2 ;;
    --force)        FORCE=true;              shift   ;;
    --keep-workdir) KEEP_WORKDIR=true;       shift   ;;
    *) die "Unknown argument: $1" ;;
  esac
done

# ── Load config ───────────────────────────────────────────────────────────────
CONFIG_FILE="${RUNNER_INSTALL_DIR}/.runner_config"
[[ -f "$CONFIG_FILE" ]] && source "$CONFIG_FILE"
RUNNER_WORKDIR="${RUNNER_WORKDIR:-/scratch/gh-runner/work}"

# ── Confirmation ──────────────────────────────────────────────────────────────
if ! $FORCE; then
  echo ""
  warn "This will de-register and remove the GitHub Actions runner at:"
  warn "  ${RUNNER_INSTALL_DIR}"
  echo ""
  read -rp "Continue? [y/N] " CONFIRM
  [[ "$CONFIRM" =~ ^[Yy]$ ]] || { info "Aborted."; exit 0; }
fi

# ── Stop and disable systemd service ─────────────────────────────────────────
info "Stopping systemd service '${SERVICE_NAME}'..."
if systemctl is-active --quiet "${SERVICE_NAME}" 2>/dev/null; then
  if [[ $EUID -eq 0 ]]; then
    systemctl stop    "${SERVICE_NAME}" || warn "Failed to stop service."
    systemctl disable "${SERVICE_NAME}" || warn "Failed to disable service."
    rm -f "/etc/systemd/system/${SERVICE_NAME}.service"
    systemctl daemon-reload
  elif sudo -n true 2>/dev/null; then
    sudo systemctl stop    "${SERVICE_NAME}" || warn "Failed to stop service."
    sudo systemctl disable "${SERVICE_NAME}" || warn "Failed to disable service."
    sudo rm -f "/etc/systemd/system/${SERVICE_NAME}.service"
    sudo systemctl daemon-reload
  else
    warn "Cannot stop systemd service without sudo. Stop it manually:"
    warn "  sudo systemctl stop ${SERVICE_NAME}"
  fi
  success "Systemd service stopped."
else
  info "Service '${SERVICE_NAME}' not running — skipping."
fi

# ── Remove cron job ───────────────────────────────────────────────────────────
info "Removing poller cron job..."
( crontab -l 2>/dev/null | grep -v "runner_poller.sh" ) | crontab - 2>/dev/null || true
success "Cron job removed."

# ── Cancel any pending SLURM ephemeral jobs ───────────────────────────────────
if command -v scancel &>/dev/null; then
  info "Cancelling SLURM ephemeral runner jobs..."
  EPHEMERAL_JOB_IDS=$(squeue --name "gh-ephemeral" --noheader --format="%i" 2>/dev/null || echo "")
  if [[ -n "$EPHEMERAL_JOB_IDS" ]]; then
    echo "$EPHEMERAL_JOB_IDS" | xargs scancel 2>/dev/null || warn "Some jobs could not be cancelled."
    success "SLURM jobs cancelled."
  else
    info "No ephemeral SLURM jobs found."
  fi
fi

# ── Fetch removal token from API if PAT available ────────────────────────────
if [[ -z "$REMOVAL_TOKEN" ]] && [[ -n "${GH_PAT:-}" ]] && [[ -n "${GH_REPO_URL:-}" ]]; then
  info "Fetching removal token from GitHub API..."
  GH_ORG_REPO=$(echo "${GH_REPO_URL}" | sed 's|https://github.com/||')
  REMOVAL_TOKEN=$(curl -sSf -X POST \
    -H "Authorization: Bearer ${GH_PAT}" \
    -H "Accept: application/vnd.github+json" \
    -H "X-GitHub-Api-Version: 2022-11-28" \
    "https://api.github.com/repos/${GH_ORG_REPO}/actions/runners/remove-token" \
    2>/dev/null | python3 -c "import sys,json; print(json.load(sys.stdin).get('token',''))" || echo "")
  if [[ -n "$REMOVAL_TOKEN" ]]; then
    success "Removal token obtained."
  else
    warn "Could not obtain removal token via API."
  fi
fi

# ── De-register runner ────────────────────────────────────────────────────────
if [[ -f "${RUNNER_INSTALL_DIR}/config.sh" ]]; then
  info "De-registering runner from GitHub..."
  cd "${RUNNER_INSTALL_DIR}"
  if [[ -n "$REMOVAL_TOKEN" ]]; then
    ./config.sh remove --unattended --token "$REMOVAL_TOKEN" 2>/dev/null \
      && success "Runner de-registered." \
      || warn "De-registration failed (token may have expired). Remove manually in GitHub repo Settings → Actions → Runners."
  else
    warn "No removal token — runner will appear offline in GitHub."
    warn "Remove it manually: Repo Settings → Actions → Runners → Remove."
  fi
else
  warn "Runner binary not found at ${RUNNER_INSTALL_DIR}/config.sh — skipping de-registration."
fi

# ── Remove ephemeral runner directories ───────────────────────────────────────
EPHEMERAL_DIR="${RUNNER_INSTALL_DIR}/ephemeral"
if [[ -d "$EPHEMERAL_DIR" ]]; then
  info "Removing ephemeral runner directories..."
  rm -rf "$EPHEMERAL_DIR"
  success "Ephemeral dirs removed."
fi

# ── Remove log files ──────────────────────────────────────────────────────────
info "Removing log files..."
rm -f "${RUNNER_INSTALL_DIR}"/*.log 2>/dev/null || true

# ── Remove work directory ─────────────────────────────────────────────────────
if ! $KEEP_WORKDIR && [[ -d "$RUNNER_WORKDIR" ]]; then
  info "Removing work directory: ${RUNNER_WORKDIR}"
  rm -rf "$RUNNER_WORKDIR" || warn "Could not remove ${RUNNER_WORKDIR} (may need elevated permissions)."
  success "Work directory removed."
fi

# ── Remove installation directory ─────────────────────────────────────────────
info "Removing runner installation directory: ${RUNNER_INSTALL_DIR}"
rm -rf "${RUNNER_INSTALL_DIR}" || {
  warn "Could not fully remove ${RUNNER_INSTALL_DIR}."
  warn "Remove manually: rm -rf ${RUNNER_INSTALL_DIR}"
}

success "Teardown complete. Runner '${GH_RUNNER_NAME:-unknown}' removed."
echo ""
echo -e "${CYAN}To re-install, run:${RESET}"
echo "  GH_RUNNER_TOKEN=<new_token> GH_REPO_URL=<repo_url> bash scripts/setup_runner.sh"
echo ""
